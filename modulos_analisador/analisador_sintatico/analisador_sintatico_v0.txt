
PROGRAMA =
    <DECLARACAO>
    <DECLARACAO> <PROGRAMA>
    <>

DECLARACAO =
    <DECLARACAO_VARIAVEL>
    <DECLARACAO_FUNCAO>
    <CONDICIONAIS>
    <LOOP>
    <RETORNO>
    <ATRIBUICAO>

DECLARACAO_VARIAVEL = 
    <TYPE> <IDENTIFIER> = <EXPRESSAO>;

EXPRESSAO = 
    <EXPRESSAO_ARITMETICA>
    <EXPRESSAO_LOGICA>
    <CHAMADA_FUNCAO>
    <ATRIBUICAO>
    <VALOR_PRIMITIVO>

EXPRESSAO_ARITMETICA =
    <VALOR_PRIMITIVO> <OPERATOR> <VALOR_PRIMITIVO>
    <VALOR_PRIMITIVO> <OPERATOR> <EXPRESSAO_ARITMETICA>
    ( <EXPRESSAO_ARITMETICA> )

EXPRESSAO_LOGICA =
    <VALOR_PRIMITIVO> <OPERATOR> <VALOR_PRIMITIVO>
    ( <EXPRESSAO_LOGICA> )
    <BOOLEAN>

ATRIBUICAO = <IDENTIFIER> = <EXPRESSAO>

VALOR_PRIMITIVO =
    <IDENTIFIER>
    <INTEGER>
    <STRING>
    <BOOLEAN>

CONDICIONAIS =
    if ( <EXPRESSAO_LOGICA> ) { <PROGRAMA> }
    if ( <EXPRESSAO_LOGICA> ) { <PROGRAMA> } else { <PROGRAMA> }

LOOP =
    while ( <EXPRESSAO_LOGICA> ) { <PROGRAMA> }

RETORNO =
    return <EXPRESSAO>;

PARAMETRO =
    <TYPE> <IDENTIFIER>
    <TYPE> <IDENTIFIER>, <PARAMETRO>
    <>

CHAMADA_PARAMETRO =
    <EXPRESSAO>
    <EXPRESSAO>, <CHAMADA_PARAMETRO>
    <>

DECLARACAO_FUNCAO =
    <TYPE> function <IDENTIFIER> (<PARAMETRO>) { <PROGRAMA> }

CHAMADA_FUNCAO =
    <IDENTIFIER>(<PARAMETRO>)
